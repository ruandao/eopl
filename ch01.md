  (define swapper
    (lambda (s1 s2 slist)
      (if (null? slist)
            '()
            (if (symbol? (car slist))
                (cons (swapper-item s1 s2 (car slist))
                      (swapper s1 s2 (cdr slist)))
                (cons (swapper s1 s2 (car slist))
                      (swapper s1 s2 (cdr slist)))))))
  (define swapper-item
    (lambda (s1 s2 s)
      (cond ((= s1 s) s2)
            ((= s2 s) s1)
            (else s)))))
