  (define (zero)
    '())
  (define (successor n)
    (if (null? n)
        '(1)
        (if (= N (+ 1 (car n)))
          (cons 0 (successor (cdr n)))
          (cons (+ 1 (car n)) (cdr n ))))))
  (define (predecessor n)
    (if (null? n)
        (report-error)
        (if (= 0 (car n))
            (cons (- N 1) (predecessor (cdr n)))
            (cons (- (car n) 1) (cdr n)))))
  (define (is-zero? n)
    (null? n))
  (define (mul n m)
    (if (is-zero? m)
        (zero)
        (plus n (mul n (predecessor m)))))
  (define (factorial n)
    (if (is-zero (predecessor n))))

  (define empty-stack
    (lambda ()
      '()))
  (define (empty-stack? s)
    (null? s))
  (define (push s v)
    (cons v s))
  (define (pop s)
    (cdr s))
  (define (top s)
    (car s))
  observers: empty-stack?, top
  constructors: empty-stack, push, pop
